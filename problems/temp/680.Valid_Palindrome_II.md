## Problem Description

Given a string `s`, return `true` *if the `s` can be palindrome after deleting **at most one** character from it.*

 

**Example 1:**

**Input:** s = "aba"  <br>
**Output:** true  <br>

**Example 2:**

**Input:** s = "abca"  <br>
**Output:** true  <br>
**Explanation:** You could delete the character 'c'.

class Solution:
    def validPalindrome(self, s: str) -> bool:
        n = len(s)
        l = 0
        r = n - 1

        while l < r:
            if s[l] != s[r]:
                s1=s[0:l]+s[l+1:n]
                s2=s[0:r]+s[r+1:n]
                if s1 == s1[::-1] or s2 == s2[::-1]:
                    return True
                else: return False      
            l += 1
            r -= 1
        
        return True
