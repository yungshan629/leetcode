    def jump(self, nums: List[int]) -> int:
        move = 0
        n = len(nums)
        if n == 1:
            return 0


        i = 0
        while i + nums[i] < n-1:
            list = nums[i+1:i+nums[i]+1]
            if len(set(list)) == 1:
                step = nums[i]
            else:
                step = list.index(max(list))+ 1
   

            i += step
            move += 1
        

        return move + 1
